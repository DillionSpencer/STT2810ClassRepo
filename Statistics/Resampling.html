set.seed(123)
sims <- 10000 - 1
meandiff <- numeric(sims)
for(i in 1:sims){
  ans <- tapply(Fschis$worms, sample(Fschis$group), mean)
  meandiff[i] <- ans[1] -ans[2]
}
pvalue <- (sum(meandiff > obsdiff) + 1)/(sims + 1)
pvalue
[1] 0.0034
